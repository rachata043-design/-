<script>
function calculate() {
  let totalFine = 0, totalJail = 0;

  document.querySelectorAll('input[type=number][data-fine]').forEach(el => {
    let qty = parseInt(el.value) || 0;
    totalFine += parseInt(el.dataset.fine) * qty;
    totalJail += parseInt(el.dataset.jail) * qty;
  });

  document.querySelectorAll('input[type=checkbox]:checked').forEach(el => {
    if(el.dataset.multiplier){
      let multi = parseInt(el.dataset.multiplier);
      totalFine *= multi;
      totalJail += parseInt(el.dataset.jail || 0);
    } else if(el.dataset.jailMultiplier){
      let multi = parseInt(el.dataset.jailMultiplier);
      totalFine *= 2;
      totalJail *= multi;
    } else {
      totalFine += parseInt(el.dataset.fine);
      totalJail += parseInt(el.dataset.jail);
    }
  });

  let redMoney = parseInt(document.getElementById('redMoney').value) || 0;
  let redNotice = '';
  if(redMoney >= 500){ totalFine += redMoney*5; totalJail += 15; }
  else if(redMoney>0){ redNotice='⚠️ เงินแดงต่ำกว่า 500: ยึดอย่างเดียว ไม่ต้องปรับ'; }

  let reduceJail = totalJail - 5; if(reduceJail<0) reduceJail=0;
  let bail = reduceJail*300;
  let remain = totalJail>5?5:totalJail;

  document.getElementById('fine').innerText = totalFine.toLocaleString();
  document.getElementById('jail').innerText = totalJail;
  document.getElementById('bail').innerText = bail.toLocaleString();
  document.getElementById('remain').innerText = remain;
  document.getElementById('total').innerText = (totalFine+bail).toLocaleString();
  document.getElementById('redNotice').innerText = redNotice;
}

function resetForm(){
  document.querySelectorAll('input[type=number]').forEach(el=>el.value=0);
  document.querySelectorAll('input[type=checkbox]').forEach(el=>el.checked=false);
  document.getElementById('redMoney').value='';
  document.getElementById('fine').innerText='0';
  document.getElementById('jail').innerText='0';
  document.getElementById('bail').innerText='0';
  document.getElementById('total').innerText='0';
  document.getElementById('remain').innerText='0';
  document.getElementById('redNotice').innerText='';
}
</script>
